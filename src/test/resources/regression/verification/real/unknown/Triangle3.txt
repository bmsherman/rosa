import leon.Real
import Real._


object Triangle3 {

  def triangle(a: Real, b: Real, c: Real): Real = {
    require(a.in(1.0, 9.0) && b.in(1.0, 9.0) && c.in(1.0, 9.0) &&
      a + b > c + 1e-9 && a + c > b + 1e-9 && b + c > a + 1e-9)

    val s = (a + b + c)/2.0
    sqrt(s * (s - a) * (s - b) * (s - c))
  } ensuring(res => 4.0e-5 < res && res < 35.07404 && res +/- 2.61727e-7)
}
